# MrReq

这是一个组合组件，它包含了

1. MrResource: 一个基于RESTful的ajax请求模块
2. GENE: 一种类似基因的遗传信息
3. MrProcess: MrLoader + MrNodataComponent


## Basic usage

<Playground>
    {() => {

        let demo = MrResource.pool('/assets/pie.json')
        let $pool = {demo};
        let req = {
            api: 'demo',
            method: 'get'
        }

        return (
            <MrReq req={req} pool={$pool} transmit="abc:res.data">
                {(abc) => JSON.stringify(abc)}
            </MrReq>
        );
    }}
</Playground>

## 与基因组件MrEcharts配合使用

<Playground>
    {() => {

        let demo = MrResource.pool('/assets/pie.json')
        let $pool = {demo};
        let req = {
            api: 'demo',
            method: 'get'
        }

        return (
            <section style={{height: 300}}>
                <MrReq req={req} pool={$pool} transmit="data:res.data">
                    {/**单独调用MrEcharts组件需要配置data属性*/}
                    {/**这里MrEcharts继承了MrReq的遗传基因片段：data (transmit="data:res.data) */}
                    <MrEcharts chartTypes="pie" />
                </MrReq>
            </section>
        );
    }}
</Playground>
